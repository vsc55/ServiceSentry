{# ================================================================ Config — Auth tab: SAML2 Entra ID wizard. Only the SAML2-specific pieces live here; the modal, device-code flow, poll loop, tabs and copy plumbing are the SHARED generic wizard (partials/credentials/_provision_wizard.html :: showEntraIdProvisionWizard). showEntraSaml2Wizard just parameterises it (endpoints, cert-aware manual scripts, ACS/entity info) and supplies onResult to apply the saml2| config. _entraSaml2CertEnd, _entraSaml2AzCliScript, _entraSaml2Ps1Script, showEntraSaml2Wizard, _entraSaml2Success, _entraSaml2Apply. ================================================================ #} // ── SAML2 Entra ID wizard (delegates to the generic provisioning wizard) ──────── // The default app name (ENTRA_APP_NAME_SAML2) is injected from Python // (providers.entraid SAML2_APP_NAME) via core/_constants.html — no local literal. function _entraSaml2CertEnd() { // addTokenSigningCertificate max = now + 3 years return new Date(Date.now() + 3 * 365 * 24 * 3600 * 1000).toISOString().slice(0, 19) + 'Z'; } function _entraSaml2AzCliScript(acsUrl, entityId, appName = ENTRA_APP_NAME_SAML2) { const certEnd = _entraSaml2CertEnd(); return `#!/bin/bash # ${t('entra_script_requires')}: az CLI (https://docs.microsoft.com/cli/azure/install-azure-cli) # ${t('entra_script_run_first')}: az login APP_NAME="${appName}" ACS_URL="${acsUrl}" SP_ENTITY_ID="${entityId}" # 1 — ${t('entra_saml2_step1')} APP=$(az ad app create \\ --display-name "$APP_NAME" \\ --sign-in-audience AzureADMyOrg \\ --output json) OBJECT_ID=$(echo "$APP" | python3 -c "import sys,json; print(json.load(sys.stdin)['id'])") CLIENT_ID=$(echo "$APP" | python3 -c "import sys,json; print(json.load(sys.stdin)['appId'])") TENANT_ID=$(az account show --query tenantId -o tsv) # 1b — ${t('entra_saml2_step1b')} az rest --method PATCH \\ --url "https://graph.microsoft.com/v1.0/applications/$OBJECT_ID" \\ --headers "Content-Type=application/json" \\ --body '{"identifierUris":["'"$SP_ENTITY_ID"'"],"groupMembershipClaims":"SecurityGroup","requiredResourceAccess":[{"resourceAppId":"00000003-0000-0000-c000-000000000000","resourceAccess":[{"id":"5b567255-7703-4780-807c-7be8301ae99b","type":"Role"}]}]}' # 2 — ${t('entra_saml2_step2')} SP=$(az rest --method POST \\ --url "https://graph.microsoft.com/v1.0/servicePrincipals" \\ --headers "Content-Type=application/json" \\ --body '{"appId":"'"$CLIENT_ID"'","tags":["WindowsAzureActiveDirectoryIntegratedApp"]}') SP_ID=$(echo "$SP" | python3 -c "import sys,json; print(json.load(sys.stdin)['id'])") az rest --method PATCH \\ --url "https://graph.microsoft.com/v1.0/servicePrincipals/$SP_ID" \\ --headers "Content-Type=application/json" \\ --body '{"preferredSingleSignOnMode":"saml"}' # 2b — ${t('entra_saml2_step2b')} az rest --method PATCH \\ --url "https://graph.microsoft.com/v1.0/servicePrincipals/$SP_ID" \\ --headers "Content-Type=application/json" \\ --body '{"replyUrls":["'"$ACS_URL"'"]}' # 3 — ${t('entra_saml2_step3')} CERT_JSON=$(az rest --method POST \\ --url "https://graph.microsoft.com/v1.0/servicePrincipals/$SP_ID/addTokenSigningCertificate" \\ --headers "Content-Type=application/json" \\ --body '{"displayName":"CN='"$APP_NAME"'","endDateTime":"${certEnd}"}') IDP_CERT_B64=$(echo "$CERT_JSON" | python3 -c "import sys,json; print(json.load(sys.stdin)['value'])") IDP_CERT_PEM=$(echo "$IDP_CERT_B64" | python3 -c 'import sys,textwrap b=sys.stdin.read().strip() print("-----BEGIN CERTIFICATE-----") [print(b[i:i+64]) for i in range(0,len(b),64)] print("-----END CERTIFICATE-----") ') # 4 — ${t('entra_saml2_step4')} GRAPH_SP_ID=$(az ad sp show --id 00000003-0000-0000-c000-000000000000 --query id -o tsv) az rest --method POST \\ --url "https://graph.microsoft.com/v1.0/servicePrincipals/$SP_ID/appRoleAssignments" \\ --headers "Content-Type=application/json" \\ --body '{"principalId":"'"$SP_ID"'","resourceId":"'"$GRAPH_SP_ID"'","appRoleId":"5b567255-7703-4780-807c-7be8301ae99b"}' echo "" echo "IdP Entity ID: https://sts.windows.net/$TENANT_ID/" echo "IdP SSO URL: https://login.microsoftonline.com/$TENANT_ID/saml2" echo "" echo "IdP Certificate (paste in idp_cert field):" echo "$IDP_CERT_PEM"`; } function _entraSaml2Ps1Script(acsUrl, entityId, appName = ENTRA_APP_NAME_SAML2) { const certEnd = _entraSaml2CertEnd(); return `# ${t('entra_script_requires')}: Install-Module Microsoft.Graph -Scope CurrentUser Connect-MgGraph -Scopes "Application.ReadWrite.All", "AppRoleAssignment.ReadWrite.All" $appName = "${appName}" $acsUrl = "${acsUrl}" $entityId = "${entityId}" # ${t('entra_saml2_step1')} $reqAccess = @(@{ ResourceAppId = "00000003-0000-0000-c000-000000000000" ResourceAccess = @(@{ Id = "5b567255-7703-4780-807c-7be8301ae99b"; Type = "Role" }) }) $app = New-MgApplication \` -DisplayName $appName \` -SignInAudience "AzureADMyOrg" \` -IdentifierUris @($entityId) \` -GroupMembershipClaims "SecurityGroup" \` -RequiredResourceAccess $reqAccess # ${t('entra_saml2_step2')} $sp = New-MgServicePrincipal -AppId $app.AppId \` -Tags @("WindowsAzureActiveDirectoryIntegratedApp") Invoke-MgGraphRequest -Method PATCH \` -Uri "https://graph.microsoft.com/v1.0/servicePrincipals/$($sp.Id)" \` -Body @{ preferredSingleSignOnMode = "saml" } \` -ContentType "application/json" # ${t('entra_saml2_step2b')} Invoke-MgGraphRequest -Method PATCH \` -Uri "https://graph.microsoft.com/v1.0/servicePrincipals/$($sp.Id)" \` -Body @{ replyUrls = @($acsUrl) } \` -ContentType "application/json" # ${t('entra_saml2_step3')} $certResult = Invoke-MgGraphRequest -Method POST \` -Uri "https://graph.microsoft.com/v1.0/servicePrincipals/$($sp.Id)/addTokenSigningCertificate" \` -Body @{ displayName = "CN=$appName"; endDateTime = "${certEnd}" } \` -ContentType "application/json" $b64 = $certResult.value $wrapped = ($b64 -replace '.{64}', "$0\`n").TrimEnd() $idpCert = "-----BEGIN CERTIFICATE-----\`n" + $wrapped + "\`n-----END CERTIFICATE-----" # ${t('entra_saml2_step4')} $graphSp = Get-MgServicePrincipal -Filter "appId eq '00000003-0000-0000-c000-000000000000'" New-MgServicePrincipalAppRoleAssignment \` -ServicePrincipalId $sp.Id -PrincipalId $sp.Id \` -ResourceId $graphSp.Id \` -AppRoleId "5b567255-7703-4780-807c-7be8301ae99b" $tenantId = (Get-MgOrganization).Id Write-Host "IdP Entity ID: https://sts.windows.net/$tenantId/" Write-Host "IdP SSO URL: https://login.microsoftonline.com/$tenantId/saml2" Write-Host "" Write-Host "IdP Certificate (paste in idp_cert field):" Write-Host $idpCert`; } function showEntraSaml2Wizard() { const _base = publicBaseUrl(); // shared helper (core/_utils.html) const acsUrl = (configData?.saml2?.sp_acs_url || '').trim() || (_base + '/auth/saml2/acs'); const entityId = (configData?.saml2?.sp_entity_id || '').trim() || _base; // Read-only ACS URL + Entity ID shown above the manual scripts (SP details the // admin needs when registering the app by hand). const _row = (label, id, val) => `
`; showEntraIdProvisionWizard({ url: '/api/v1/auth/entraid/saml2/device-code', pollUrl: '/api/v1/auth/entraid/saml2/device-poll', title: t('entra_saml2_wizard_title'), intro: t('entra_saml2_auto_intro'), provision: { app_name: ENTRA_APP_NAME_SAML2 }, scripts: { ps1: (name) => _entraSaml2Ps1Script(acsUrl, entityId, name), azcli: (name) => _entraSaml2AzCliScript(acsUrl, entityId, name), }, manualExtraHtml: _row(t('entra_saml2_acs_url_label'), 'entraSaml2AcsInput', acsUrl) + _row(t('entra_saml2_entity_id_label'), 'entraSaml2EntityInput', entityId), onResult: _entraSaml2Success, }); } // Deep link to the app's SAML SSO blade in the Entra portal (needs the SP objectId // AND the appId). function _entraSaml2SsoUrl(spObjectId, appId) { spObjectId = (spObjectId || '').trim(); appId = (appId || '').trim(); if (!spObjectId || !appId) return ''; return 'https://entra.microsoft.com/#view/Microsoft_AAD_IAM/ManagedAppMenuBlade/~/SignOn/objectId/' + encodeURIComponent(spObjectId) + '/appId/' + encodeURIComponent(appId); } // A read-only, copyable label:value row (values the admin must paste into Azure). function _s2CopyRow(label, val) { val = (val || '').trim(); if (!val) return ''; return `
${esc(label)}
`; } // "Open in Entra ID" button for the SAML2 app → straight to its SSO blade (falls // back to the app-registration link if we only have the appId). function _entraSaml2SsoLink(data) { const url = _entraSaml2SsoUrl(data.sp_object_id, data.sp_app_id); if (!url) return (typeof _entraAppLink === 'function') ? _entraAppLink(data.sp_app_id) : ''; return ` ${esc(t('entra_open_app'))}`; } // Back-fill the Graph client secret on an EXISTING SAML2 app (registered before // secrets were provisioned) so the group→role mapping can read directory groups. // Reuses the generic wizard in "simple" mode (just a device-code sign-in). function _entraSaml2AddSecret() { showEntraIdProvisionWizard({ url: '/api/v1/auth/entraid/saml2/secret/device-code', pollUrl: '/api/v1/auth/entraid/saml2/device-poll', title: t('entra_saml2_add_secret_title'), intro: t('entra_saml2_add_secret_intro'), simple: true, onResult: (d) => { if (d.graph_secret) { updateField('config', 'saml2|graph_secret', d.graph_secret); saveConfig(); } return { html: `
${esc(t('entra_saml2_add_secret_done'))}
`, autoClose: true }; }, }); } // Called by the generic wizard on a completed SAML2 flow: apply the discovered IdP // settings to the saml2| config and render the success panel. The "Basic SAML // Configuration" (Identifier + Reply URL) has no Graph API, so the panel guides the // admin to the app's SSO blade with the exact values to paste, and opens it. function _entraSaml2Success(d) { _entraSaml2Apply(d); const appName = d.app_name || ENTRA_APP_NAME_SAML2; const spOk = d.sp_created !== false; const ssoUrl = _entraSaml2SsoUrl(d.sp_object_id, d.client_id); const warn = (msg) => `
${esc(msg)}
`; let html = `
${esc(t('entra_saml2_auto_success'))}
`; // Manual last step: Entra has no API for the Basic SAML Configuration. html += `
${esc(t('entra_saml2_finish_hint'))}
${_s2CopyRow(t('entra_saml2_entity_id_label'), d.sp_entity_id)} ${_s2CopyRow(t('entra_saml2_acs_url_label'), d.sp_acs_url)} ${ssoUrl ? `${esc(t('entra_saml2_finish_btn'))}` : ''}
`; if (!spOk) html += warn(t('entra_sp_error') + (d.sp_error ? ` (${d.sp_error})` : '')); if (d.cert_error) html += warn(t('entra_saml2_cert_error') + ` (${d.cert_error})`); if (spOk && d.consent_granted) html += `
${esc(t('entra_consent_granted'))}
`; if (spOk && !d.consent_granted) html += warn(t('entra_consent_pending')); saveConfig(); if (ssoUrl) window.open(ssoUrl, '_blank', 'noopener'); // jump to the SSO blade to finish return { html, autoClose: false }; } function _entraSaml2Apply(d) { const patch = { idp_entity_id: d.idp_entity_id || (d.tenant_id ? `https://sts.windows.net/${d.tenant_id}/` : ''), idp_sso_url: d.idp_sso_url || (d.tenant_id ? `https://login.microsoftonline.com/${d.tenant_id}/saml2` : ''), button_label: 'Microsoft Entra ID', username_attr: 'http://schemas.xmlsoap.org/ws/2005/05/identity/claims/name', email_attr: 'http://schemas.xmlsoap.org/ws/2005/05/identity/claims/emailaddress', name_attr: 'http://schemas.microsoft.com/identity/claims/displayname', groups_attr: 'http://schemas.microsoft.com/ws/2008/06/identity/claims/groups', }; if (d.idp_cert) patch.idp_cert = d.idp_cert; if (d.client_id) patch.sp_app_id = d.client_id; // "open in Entra ID" link (appId) if (d.sp_object_id) patch.sp_object_id = d.sp_object_id; // deep link to the app's SSO blade if (d.graph_secret) patch.graph_secret = d.graph_secret; // Graph reads for the group→role mapping // Authoritative Entity ID: whatever Entra actually registered (the requested one, // or the api://{appId} fallback when the domain wasn't verifiable). ServiceSentry // must send exactly this as its SAML Issuer, so overwrite the configured value. if (d.sp_entity_id) patch.sp_entity_id = d.sp_entity_id; if (d.sp_acs_url && !(configData?.saml2?.sp_acs_url || '').trim()) patch.sp_acs_url = d.sp_acs_url; for (const [field, value] of Object.entries(patch)) { updateField('config', `saml2|${field}`, value); } renderConfig(); }
