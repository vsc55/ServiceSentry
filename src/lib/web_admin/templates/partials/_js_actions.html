{# ================================================================ Actions — field updates, item add/remove, save operations. ================================================================ #} // ──────────────────────────── Actions ───────────────────────────── function markDirty(target) { const id = target === 'modules' ? 'badgeModulesDirty' : 'badgeConfigDirty'; document.getElementById(id)?.classList.remove('d-none'); } function clearDirty(target) { const id = target === 'modules' ? 'badgeModulesDirty' : 'badgeConfigDirty'; document.getElementById(id)?.classList.add('d-none'); } function updateField(target, pathStr, value) { const parts = pathStr.split('|'); setPath(target === 'config' ? configData : modulesData, parts, value); markDirty(target); } function updateArray(target, pathStr, csv) { const arr = csv.split(',').map(s => s.trim()).filter(Boolean); updateField(target, pathStr, arr); } function toggleModule(name, on) { modulesData[name].enabled = on; const card = document.getElementById('card-' + name); if (card) card.classList.toggle('enabled', on); markDirty('modules'); } function updatePrettyName(name, value) { if (value.trim()) { modulesData[name].pretty_name = value.trim(); } else { delete modulesData[name].pretty_name; } markDirty('modules'); // Update header text live const card = document.getElementById('card-' + name); if (card) { const label = card.querySelector('.card-header .fw-semibold'); if (label) label.textContent = modulePrettyName(name, modulesData[name]); } } function removeItem(target, parentPathStr, key) { _deleteItemCtx = { target, parentPathStr, key }; document.getElementById('confirmDeleteItemMsg').textContent = tf('confirm_delete_item', key); bootstrap.Modal.getOrCreateInstance(document.getElementById('confirmDeleteItemModal')).show(); } function _doDeleteItem() { if (!_deleteItemCtx) return; const { target, parentPathStr, key } = _deleteItemCtx; const parts = parentPathStr.split('|'); const parent = getPath(target === 'config' ? configData : modulesData, parts); if (parent && key in parent) { delete parent[key]; markDirty(target); target === 'modules' ? renderModules() : renderConfig(); } bootstrap.Modal.getInstance(document.getElementById('confirmDeleteItemModal')).hide(); _deleteItemCtx = null; } function renameItem(target, parentPathStr, oldKey) { _renameItemCtx = { target, parentPathStr, oldKey }; const input = document.getElementById('renameItemInput'); input.value = oldKey; const modalEl = document.getElementById('renameItemModal'); modalEl.addEventListener('shown.bs.modal', () => { input.focus(); input.select(); }, { once: true }); bootstrap.Modal.getOrCreateInstance(modalEl).show(); } function _doRenameItem() { if (!_renameItemCtx) return; const { target, parentPathStr, oldKey } = _renameItemCtx; const nk = document.getElementById('renameItemInput').value.trim(); if (!nk || nk === oldKey) { bootstrap.Modal.getInstance(document.getElementById('renameItemModal')).hide(); _renameItemCtx = null; return; } const parts = parentPathStr.split('|'); const data = target === 'modules' ? modulesData : configData; const parent = getPath(data, parts); if (!parent || typeof parent !== 'object') return; if (nk in parent) { showToast(tf('already_exists', nk), 'warning'); return; } // Copy value to new key, preserving insertion order const entries = Object.entries(parent); const idx = entries.findIndex(([k]) => k === oldKey); if (idx < 0) return; // Clear parent and re-insert in same order with new key for (const k of Object.keys(parent)) delete parent[k]; for (let i = 0; i < entries.length; i++) { const [k, v] = entries[i]; parent[i === idx ? nk : k] = v; } bootstrap.Modal.getInstance(document.getElementById('renameItemModal')).hide(); _renameItemCtx = null; markDirty(target); target === 'modules' ? renderModules() : renderConfig(); } // --- Add-item modal --- function openAddModal(target, pathStr) { _addItemCtx = { target, pathStr }; document.getElementById('newItemName').value = ''; bootstrap.Modal.getOrCreateInstance(document.getElementById('addItemModal')).show(); setTimeout(() => document.getElementById('newItemName').focus(), 400); } function confirmAddItem() { if (!_addItemCtx) return; const name = document.getElementById('newItemName').value.trim(); if (!name) return; const { target, pathStr } = _addItemCtx; const parts = pathStr.split('|'); const data = target === 'modules' ? modulesData : configData; const parent = getPath(data, parts); if (!parent || typeof parent !== 'object') return; if (name in parent) { showToast(tf('already_exists', name), 'warning'); return; } // Smart default: schema > copy from first dict sibling > true const schema = schemaFor(pathStr); const defaults = schemaDefaults(schema); let def = true; if (defaults) { def = {}; for (const [k, sv] of Object.entries(defaults)) { def[k] = Array.isArray(sv) ? [...sv] : sv; } } else { for (const v of Object.values(parent)) { if (typeof v === 'object' && v !== null && !Array.isArray(v)) { def = {}; for (const [k, val] of Object.entries(v)) { if (typeof val === 'boolean') def[k] = (k === 'enabled'); else if (typeof val === 'number') def[k] = 0; else if (typeof val === 'string') def[k] = ''; else if (Array.isArray(val)) def[k] = []; else def[k] = {}; } break; } } } parent[name] = def; bootstrap.Modal.getInstance(document.getElementById('addItemModal')).hide(); markDirty(target); target === 'modules' ? renderModules() : renderConfig(); _addItemCtx = null; } // --- Save --- async function saveModules() { const r = await apiPut('/api/modules', modulesData); if (r && r.ok) { showToast(t('modules_saved')); clearDirty('modules'); } else showToast(r?.error || t('save_error'), 'danger'); } async function saveConfig() { const r = await apiPut('/api/config', configData); if (r && r.ok) { showToast(t('config_saved')); clearDirty('config'); _serverConfigData = deepClone(configData); // update snapshot after successful save } else showToast(r?.error || t('save_error'), 'danger'); } async function reloadConfig() { const data = await apiGet('/api/config'); if (!data) return; configData = data; _serverConfigData = deepClone(configData); clearDirty('config'); renderConfig(); showToast(t('config_reloaded')); }
