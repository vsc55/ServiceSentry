{# ================================================================ Config — Auth tab: OIDC "Register in Azure" wizard glue. OIDC registration is FULLY automatic (no manual portal step), so it just parameterises the SHARED generic wizard (partials/credentials/_provision_wizard.html :: showEntraIdProvisionWizard): the OIDC app is a provisioning profile (web sign-in app + groups claim + require-assignment + openid/email/profile/User.Read + Group.Read.All). On completion the created app's tenant/client/secret + the OIDC claim mapping are written to the oidc config. (The "Open in Entra ID" helper _entraAppLink and the section renderer live in _renderers.html.) _entraRedirectUri, showEntraWizard, _entraApply. ================================================================ #} function _entraRedirectUri() { return publicBaseUrl() + '/auth/oidc/callback'; // shared helper (core/_utils.html) } function showEntraWizard() { showEntraIdProvisionWizard({ url: '/api/v1/auth/entraid/provision/device-code', provision: { app_name: ENTRA_APP_NAME_OIDC, app_roles: ['Group.Read.All'], scopes: ['openid', 'email', 'profile', 'User.Read'], redirect_uris: [_entraRedirectUri()], group_claims: true, require_assignment: true, }, }, (fields) => { _entraApply(fields || {}); if (typeof saveConfig === 'function') saveConfig(); }); } function _entraApply(d) { const patch = { username_claim: 'preferred_username', email_claim: 'email', name_claim: 'name', groups_claim: 'groups', scopes: 'openid email profile', button_label: 'Microsoft Entra ID', }; if (d.provider_url) patch.provider_url = d.provider_url; else if (d.tenant_id) patch.provider_url = `https://login.microsoftonline.com/${d.tenant_id}/v2.0`; if (d.client_id) patch.client_id = d.client_id; if (d.client_secret) patch.client_secret = d.client_secret; // updateField registers each path in _dirtyFields so saveConfig() picks them up for (const [field, value] of Object.entries(patch)) { updateField('config', `oidc|${field}`, value); } renderConfig(); }
