// ── Services tab ────────────────────────────────────────────────────────────── // Central view of the background services (scheduler / syslog / worker / DB) with // start/stop for the ones this process hosts. Status comes from /api/v1/services; // control via POST /api/v1/services//. // state → Bootstrap colour for the status badge / dot. const _SVC_STATE_CLS = { running: 'success', active: 'success', alive: 'success', embedded: 'info', stopped: 'secondary', disabled: 'secondary', external: 'info', stale: 'warning', unknown: 'secondary', down: 'danger', error: 'danger', }; // Imperative commands offered per service (shown with the services_control perm). // These work whether the service is hosted here or in a dedicated container — // the queue is drained by whichever instance owns the service. const _SVC_COMMANDS = { monitoring: ['run_now', 'reload', 'clear_status'], events: ['run_now', 'reload'], syslog: ['reload', 'prune'], }; let _svcPollTimer = null; function _svcStateLabel(state) { return t('svc_state_' + state) || state; } function _svcAgo(ts) { if (!ts) return '—'; const secs = Math.max(0, Math.floor(Date.now() / 1000 - ts)); if (secs < 60) return t('svc_just_now'); if (secs < 3600) return Math.floor(secs / 60) + ' min'; if (secs < 86400) return Math.floor(secs / 3600) + ' h'; return Math.floor(secs / 86400) + ' d'; } function _svcIn(secs) { if (secs == null) return '—'; if (secs <= 0) return t('svc_now'); if (secs < 60) return secs + ' s'; if (secs < 3600) return Math.floor(secs / 60) + ' min'; return Math.floor(secs / 3600) + ' h'; } function _svcRow(label, value) { return `
${esc(label)}${value}
`; } // Render one detail row from its backend descriptor. Each service emits its own // rows in status().detail = [{label_key, value | value_key, fmt?}], so this stays // fully generic — a new service's rows show up with no change here: // value_key → an i18n value (e.g. "Embedded"); fmt 'ago'/'in' → relative time; // otherwise the value is shown verbatim (escaped). function _svcDetailRow(r) { let v; if (r.value_key) v = esc(t(r.value_key)); else if (r.fmt === 'ago') v = _svcAgo(r.value); else if (r.fmt === 'in') v = _svcIn(r.value); else v = esc(r.value == null ? '—' : String(r.value)); return _svcRow(t(r.label_key), v); } // One live instance (heartbeat), two lines: state + host (+ leader) on top, a muted // meta line (this process · mode · version · age) below. Shows remote replicas in // another pod, not only the one this process hosts. function _svcInstanceRow(inst) { const cls = _SVC_STATE_CLS[inst.derived_state] || 'secondary'; const host = esc(inst.host || '—'); const modeK = inst.mode === 'embedded' ? 'svc_mode_embedded' : 'svc_mode_container'; const when = inst.last_cycle_at ? _svcAgo(inst.last_cycle_at) : _svcAgo(inst.last_seen); // Leader / standby badge for single-owner services (the heartbeat detail // carries `leader` only for leader-gated services; absent for active-active). let lead = ''; const lv = inst.detail && inst.detail.leader; if (typeof lv === 'boolean') { lead = lv ? `${esc(t('svc_leader'))}` : `${esc(t('svc_standby'))}`; } // Muted meta line parts. const parts = []; if (inst.is_self) parts.push(esc(t('svc_self_tag'))); parts.push(esc(t(modeK))); if (inst.version) parts.push(esc(inst.version)); parts.push(when); return `
${esc(_svcStateLabel(inst.derived_state))}${host} ${lead}
${parts.join(' · ')}
`; } // Render one service card. `meta` carries the backend entry verbatim. function _svcCard(meta) { const cls = _SVC_STATE_CLS[meta.state] || 'secondary'; const canCtl = currentUser.permissions?.includes('services_control'); // Primary action: start/stop (only for services this process hosts embedded). let primary = ''; if (meta.controllable && canCtl) { primary = meta.running ? `` : ``; } // Hint for services a dedicated container owns (no start/stop from here). const hint = (!meta.embedded) ? `${esc(t('svc_external_managed'))}` : ''; // Imperative commands behind a "⋯" menu — queued for whichever instance hosts // the service (here or a remote pod), so they declutter the card. let cmdMenu = ''; const cmds = _SVC_COMMANDS[meta.key]; if (cmds && canCtl) { const items = cmds.map(a => `
    • `).join(''); cmdMenu = `
      ${items}
    `; } // Live instances (heartbeat) — set apart from the service-info rows above by a // stronger divider + an uppercase section label. let instancesHtml = ''; if (Array.isArray(meta.instances) && meta.instances.length) { instancesHtml = `
    ${esc(t('svc_instances'))} (${meta.instances.length})
    ${meta.instances.map(_svcInstanceRow).join('')}
    `; } // Per-card refresh (bottom-left) — only for the actual services; the database // views are static, so refreshing one card makes no sense there. const showRefresh = !String(meta.key).startsWith('database'); const refreshBtn = showRefresh ? `` : ''; const left = `${refreshBtn}${hint ? `${hint}` : ''}`; const right = `${cmdMenu}${primary}`; const footer = (left || right) ? `
    ${left} ${right}
    ` : ''; return `
    ${esc(meta.title)}
    ${esc(_svcStateLabel(meta.state))}
    ${meta.rows.join('')}${instancesHtml}
    ${footer}
    `; } // Build the _svcCard meta from a backend service entry (shared by the full render // and the per-card refresh). function _svcMeta(key, d) { return { key, icon: d.icon || 'bi-gear', title: d.label_key ? t(d.label_key) : key, state: d.state, running: !!d.running, controllable: !!d.controllable, embedded: !!d.embedded, rows: (d.detail || []).map(_svcDetailRow), instances: d.instances || [], }; } // Force-refresh ONLY one card: fetch fresh data now and replace just that node. async function refreshServiceCard(key, btn) { const ico = btn && btn.querySelector('i'); if (ico) ico.classList.add('ss-spin'); const resp = await apiGet('/api/v1/services'); const s = (resp && resp.services) || {}; const el = document.getElementById('svc-card-' + key); if (!(key in s)) { if (el) el.remove(); return; } // service gone → drop card const html = _svcCard(_svcMeta(key, s[key])); if (el) { el.outerHTML = html; } else { renderServices(); } } async function renderServices() { const c = document.getElementById('services-container'); if (!c) return; const resp = await apiGet('/api/v1/services'); const s = (resp && resp.services) || {}; // Render one card per registered service, in registry order. Identity // (title/icon) + state/controllability come from the backend entry; the // service-specific detail rows from _SVC_DETAIL (or none for an unknown key). const cards = Object.entries(s).map(([key, d]) => _svcCard({ key, icon: d.icon || 'bi-gear', title: d.label_key ? t(d.label_key) : key, state: d.state, running: !!d.running, controllable: !!d.controllable, embedded: !!d.embedded, rows: (d.detail || []).map(_svcDetailRow), instances: d.instances || [], })); const refreshBtn = ``; c.innerHTML = `
    ${esc(t('tab_services'))}
    ${refreshBtn}
    ${cards.join('') || `

    ${esc(t('svc_none'))}

    `}
    `; // Light auto-refresh while the Services tab stays visible. if (_svcPollTimer) clearTimeout(_svcPollTimer); _svcPollTimer = setTimeout(() => { if (document.getElementById('tab-services')?.classList.contains('active')) renderServices(); }, 5000); } async function servicesControl(name, action) { const r = await apiPost(`/api/v1/services/${name}/${action}`, {}); if (r && (r.status === 200) && r.data?.ok) { showToast(t('svc_' + action + '_ok'), 'success'); } else { const reason = r?.data?.reason; showToast(reason ? (t('svc_err_' + reason) || reason) : t('svc_action_error'), 'warning'); } renderServices(); } async function servicesCommand(name, action) { const r = await apiPost(`/api/v1/services/${name}/command/${action}`, {}); if (r && (r.status === 200) && r.data?.ok) { showToast(t('svc_command_ok'), 'success'); } else { const reason = r?.data?.reason; showToast(reason ? (t('svc_err_' + reason) || reason) : t('svc_command_error'), 'warning'); } renderServices(); }
