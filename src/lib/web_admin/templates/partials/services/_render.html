// ── Services tab ────────────────────────────────────────────────────────────── // Central view of the background services (scheduler / syslog / worker / DB) with // start/stop for the ones this process hosts. Status comes from /api/v1/services; // control via POST /api/v1/services//. // state → Bootstrap colour for the status badge / dot. const _SVC_STATE_CLS = { running: 'success', active: 'success', embedded: 'info', stopped: 'secondary', disabled: 'secondary', external: 'info', stale: 'warning', unknown: 'secondary', error: 'danger', }; let _svcPollTimer = null; function _svcStateLabel(state) { return t('svc_state_' + state) || state; } function _svcAgo(ts) { if (!ts) return '—'; const secs = Math.max(0, Math.floor(Date.now() / 1000 - ts)); if (secs < 60) return t('svc_just_now'); if (secs < 3600) return Math.floor(secs / 60) + ' min'; if (secs < 86400) return Math.floor(secs / 3600) + ' h'; return Math.floor(secs / 86400) + ' d'; } function _svcIn(secs) { if (secs == null) return '—'; if (secs <= 0) return t('svc_now'); if (secs < 60) return secs + ' s'; if (secs < 3600) return Math.floor(secs / 60) + ' min'; return Math.floor(secs / 3600) + ' h'; } function _svcRow(label, value) { return `
${esc(label)}${value}
`; } // Render one service card. `meta` = {key, icon, title, rows[], canControl, running} function _svcCard(meta) { const cls = _SVC_STATE_CLS[meta.state] || 'secondary'; const canCtl = currentUser.permissions?.includes('services_control'); let actions = ''; if (meta.controllable && canCtl) { actions = meta.running ? `` : ``; } else if (!meta.embedded) { actions = `${esc(t('svc_external_managed'))}`; } return `
${esc(meta.title)}
${esc(_svcStateLabel(meta.state))}
${meta.rows.join('')}
${actions}
`; } async function renderServices() { const c = document.getElementById('services-container'); if (!c) return; const resp = await apiGet('/api/v1/services'); const s = (resp && resp.services) || {}; const cards = []; // Scheduler (embedded monitoring loop) if (s.scheduler) { const d = s.scheduler; cards.push(_svcCard({ key: 'scheduler', icon: 'bi-arrow-repeat', title: t('svc_scheduler'), state: d.state, running: d.running, controllable: d.controllable, embedded: d.embedded, rows: [ _svcRow(t('svc_interval'), d.interval != null ? d.interval + ' s' : '—'), _svcRow(t('svc_next_run'), d.running ? _svcIn(d.next_in) : '—'), _svcRow(t('svc_last_run'), _svcAgo(d.last_run)), _svcRow(t('svc_autostart'), d.autostart ? t('svc_yes') : t('svc_no')), ], })); } // Syslog receiver if (s.syslog) { const d = s.syslog; cards.push(_svcCard({ key: 'syslog', icon: 'bi-hdd-stack', title: t('svc_syslog'), state: d.state, running: d.running, controllable: d.controllable, embedded: d.embedded, rows: [ _svcRow(t('svc_ports'), `UDP ${d.udp_port||'—'} · TCP ${d.tcp_port||'—'} · TLS ${d.tls_port||'—'}`), _svcRow(t('svc_messages'), d.count), _svcRow(t('svc_mode'), d.embedded ? t('svc_mode_embedded') : t('svc_mode_container')), ], })); } // Worker (external monitoring process) if (s.worker) { const d = s.worker; cards.push(_svcCard({ key: 'worker', icon: 'bi-cpu', title: t('svc_worker'), state: d.state, running: false, controllable: false, embedded: false, rows: [ _svcRow(t('svc_last_activity'), _svcAgo(d.last_activity)), _svcRow(t('svc_recent_checks'), d.fresh ? t('svc_yes') : t('svc_no')), ], })); } // Database (system) if (s.database) { const d = s.database; cards.push(_svcCard({ key: 'database', icon: 'bi-database', title: t('svc_database'), state: d.state, running: false, controllable: false, embedded: false, rows: [ _svcRow(t('svc_driver'), esc(d.driver || '—')), _svcRow(t('svc_host'), esc(d.host || '—')), _svcRow(t('svc_db_name'), esc(d.name || '—')), ], })); } // Dedicated syslog database (only present when syslog uses its own DB) if (s.database_syslog) { const d = s.database_syslog; const rows = [ _svcRow(t('svc_driver'), esc(d.driver || '—')), _svcRow(t('svc_host'), esc(d.host || '—')), _svcRow(t('svc_db_name'), esc(d.name || '—')), ]; if (d.fell_back) rows.push(_svcRow(t('svc_status'), esc(t('svc_db_fell_back')))); cards.push(_svcCard({ key: 'database_syslog', icon: 'bi-database-fill-gear', title: t('svc_database_syslog'), state: d.state, running: false, controllable: false, embedded: false, rows, })); } const refreshBtn = ``; c.innerHTML = `
${esc(t('tab_services'))}
${refreshBtn}
${cards.join('') || `

${esc(t('svc_none'))}

`}
`; // Light auto-refresh while the Services tab stays visible. if (_svcPollTimer) clearTimeout(_svcPollTimer); _svcPollTimer = setTimeout(() => { if (document.getElementById('tab-services')?.classList.contains('active')) renderServices(); }, 5000); } async function servicesControl(name, action) { const r = await apiPost(`/api/v1/services/${name}/${action}`, {}); if (r && (r.status === 200) && r.data?.ok) { showToast(t('svc_' + action + '_ok'), 'success'); } else { const reason = r?.data?.reason; showToast(reason ? (t('svc_err_' + reason) || reason) : t('svc_action_error'), 'warning'); } renderServices(); }
