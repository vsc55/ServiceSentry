// ── Services tab ────────────────────────────────────────────────────────────── // Central view of the background services (scheduler / syslog / worker / DB) with // start/stop for the ones this process hosts. Status comes from /api/v1/services; // control via POST /api/v1/services//. // state → Bootstrap colour for the status badge / dot. const _SVC_STATE_CLS = { running: 'success', active: 'success', alive: 'success', embedded: 'info', stopped: 'secondary', disabled: 'secondary', external: 'info', stale: 'warning', unknown: 'secondary', down: 'danger', error: 'danger', }; // Imperative commands offered per service (shown with the services_control perm). // These work whether the service is hosted here or in a dedicated container — // the queue is drained by whichever instance owns the service. const _SVC_COMMANDS = { monitoring: ['run_now', 'reload', 'clear_status'], events: ['run_now', 'reload'], syslog: ['reload', 'prune'], }; let _svcPollTimer = null; function _svcStateLabel(state) { return t('svc_state_' + state) || state; } function _svcAgo(ts) { if (!ts) return '—'; const secs = Math.max(0, Math.floor(Date.now() / 1000 - ts)); if (secs < 60) return t('svc_just_now'); if (secs < 3600) return Math.floor(secs / 60) + ' min'; if (secs < 86400) return Math.floor(secs / 3600) + ' h'; return Math.floor(secs / 86400) + ' d'; } function _svcIn(secs) { if (secs == null) return '—'; if (secs <= 0) return t('svc_now'); if (secs < 60) return secs + ' s'; if (secs < 3600) return Math.floor(secs / 60) + ' min'; return Math.floor(secs / 3600) + ' h'; } // Formatted value of a backend detail descriptor {label_key, value|value_key, fmt}: // value_key → an i18n value (e.g. "Embedded"); fmt 'ago'/'in' → relative time; // otherwise the value verbatim (escaped). Returns '—' for empty. function _svcDetailValue(r) { if (r.value_key) return esc(t(r.value_key)); if (r.fmt === 'ago') return _svcAgo(r.value); if (r.fmt === 'in') return _svcIn(r.value); return esc(r.value == null ? '—' : String(r.value)); } // Service icon in a tinted square (state colour). Size/layout live in CSS // (.ss-svc-av[-lg]); only the colour + icon are dynamic here. function _svcAvatar(meta, cls, big) { return ``; } // One live instance (heartbeat), two lines: state + host (+ leader) on top, a muted // meta line (this process · mode · version · age) below. Shows remote replicas in // another pod, not only the one this process hosts. function _svcInstanceRow(inst) { const cls = _SVC_STATE_CLS[inst.derived_state] || 'secondary'; const host = esc(inst.host || '—'); const modeK = inst.mode === 'embedded' ? 'svc_mode_embedded' : 'svc_mode_container'; const when = inst.last_cycle_at ? _svcAgo(inst.last_cycle_at) : _svcAgo(inst.last_seen); // Leader / standby badge for single-owner services (the heartbeat detail // carries `leader` only for leader-gated services; absent for active-active). let lead = ''; const lv = inst.detail && inst.detail.leader; if (typeof lv === 'boolean') { lead = lv ? `${esc(t('svc_leader'))}` : `${esc(t('svc_standby'))}`; } // Meta as small pill chips (consistent with the card details) instead of a // flat dot-separated line: this process / mode / version / age. const metaChip = (icon, text, cls) => `${text}`; const metaChips = []; if (inst.is_self) metaChips.push(metaChip('bi-pc-display', esc(t('svc_self_tag')), 'bg-primary-subtle text-primary-emphasis')); metaChips.push(metaChip(inst.mode === 'embedded' ? 'bi-box-seam' : 'bi-hdd-network', esc(t(modeK)))); if (inst.version) metaChips.push(metaChip('bi-tag', 'v' + esc(inst.version))); metaChips.push(metaChip('bi-clock-history', when)); // Where this instance points: shown only for remote instances (embedded ones // share the web admin's DB, already on its Database card). `detail.db` = {main, // syslog?} = {driver, host, name}. let dbLine = ''; const db = inst.detail && inst.detail.db; if (db) { const fmt = x => x ? esc([x.driver, x.host, x.name].filter(Boolean).join(' · ')) : ''; const rows = []; if (db.main) rows.push(`${fmt(db.main)}`); if (db.syslog) rows.push(`${fmt(db.syslog)}`); if (rows.length) dbLine = `
${rows.join('·')}
`; } return `
${esc(_svcStateLabel(inst.derived_state))}${host} ${lead}
${metaChips.join('')}
${dbLine}
`; } // Live instances (heartbeat) block — shared by every card style. function _svcInstancesHtml(meta) { if (!Array.isArray(meta.instances) || !meta.instances.length) return ''; return `
${esc(t('svc_instances'))} (${meta.instances.length})
${meta.instances.map(_svcInstanceRow).join('')}
`; } // Footer (hint · ⋯ commands · start/stop) — shared by every card style. // (No per-card refresh: the Services header already has a global Refresh.) function _svcFooter(meta) { const canCtl = currentUser.permissions?.includes('services_control'); let primary = ''; if (meta.controllable && canCtl) { primary = meta.running ? `` : ``; } const hint = (!meta.embedded) ? `${esc(t('svc_external_managed'))}` : ''; let cmdMenu = ''; const cmds = _SVC_COMMANDS[meta.key]; if (cmds && canCtl) { const items = cmds.map(a => `
    • `).join(''); cmdMenu = `
      ${items}
    `; } const right = `${cmdMenu}${primary}`; return (hint || right) ? `
    ${hint} ${right}
    ` : ''; } // Card shell (column + bordered card + accent) shared by the three styles. function _svcShell(meta, cls, body) { return `
    ${body}
    `; } // Render one service card: avatar + title header, a coloured status banner and the // service details as inline chips, then the live instances and the action footer. function _svcCard(meta) { const cls = _SVC_STATE_CLS[meta.state] || 'secondary'; const chips = meta.detail.map(r => ` ${esc(t(r.label_key))} ${_svcDetailValue(r)}`).join(' '); // Solid status banner so the state pops (a faint tint didn't stand out, esp. // gray "stopped"). Yellow/cyan need dark text; the rest read on white. const onLight = (cls === 'warning' || cls === 'info'); const header = `
    ${_svcAvatar(meta, cls, false)}
    ${esc(meta.title)}
    ${esc(_svcStateLabel(meta.state))}
    ${chips ? `
    ${chips}
    ` : ''}`; const body = `${header}
    ${_svcInstancesHtml(meta)}
    ${_svcFooter(meta)}`; return _svcShell(meta, cls, body); } // Build the _svcCard meta from a backend service entry (shared by the full render // and the per-card refresh). function _svcMeta(key, d) { return { key, icon: d.icon || 'bi-gear', title: d.label_key ? t(d.label_key) : key, state: d.state, running: !!d.running, controllable: !!d.controllable, embedded: !!d.embedded, detail: d.detail || [], instances: d.instances || [], }; } async function renderServices() { const c = document.getElementById('services-container'); if (!c) return; const resp = await apiGet('/api/v1/services'); const s = (resp && resp.services) || {}; // Render one card per registered service, in registry order. Identity // (title/icon) + state/controllability come from the backend entry; the // service-specific detail rows from _SVC_DETAIL (or none for an unknown key). const cards = Object.entries(s).map(([key, d]) => _svcCard(_svcMeta(key, d))); const refreshBtn = ``; c.innerHTML = `
    ${esc(t('tab_services'))}
    ${refreshBtn}
    ${cards.join('') || `

    ${esc(t('svc_none'))}

    `}
    `; // Light auto-refresh while the Services tab stays visible. if (_svcPollTimer) clearTimeout(_svcPollTimer); _svcPollTimer = setTimeout(() => { if (document.getElementById('tab-services')?.classList.contains('active')) renderServices(); }, 5000); } async function servicesControl(name, action) { const r = await apiPost(`/api/v1/services/${name}/${action}`, {}); if (r && (r.status === 200) && r.data?.ok) { showToast(t('svc_' + action + '_ok'), 'success'); } else { const reason = r?.data?.reason; showToast(reason ? (t('svc_err_' + reason) || reason) : t('svc_action_error'), 'warning'); } renderServices(); } async function servicesCommand(name, action) { const r = await apiPost(`/api/v1/services/${name}/command/${action}`, {}); if (r && (r.status === 200) && r.data?.ok) { showToast(t('svc_command_ok'), 'success'); } else { const reason = r?.data?.reason; showToast(reason ? (t('svc_err_' + reason) || reason) : t('svc_command_error'), 'warning'); } renderServices(); }
