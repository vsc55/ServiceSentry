// ── Services tab ────────────────────────────────────────────────────────────── // Central view of the background services (scheduler / syslog / worker / DB) with // start/stop for the ones this process hosts. Status comes from /api/v1/services; // control via POST /api/v1/services//. // state → Bootstrap colour for the status badge / dot. const _SVC_STATE_CLS = { running: 'success', active: 'success', embedded: 'info', stopped: 'secondary', disabled: 'secondary', external: 'info', stale: 'warning', unknown: 'secondary', error: 'danger', }; let _svcPollTimer = null; function _svcStateLabel(state) { return t('svc_state_' + state) || state; } function _svcAgo(ts) { if (!ts) return '—'; const secs = Math.max(0, Math.floor(Date.now() / 1000 - ts)); if (secs < 60) return t('svc_just_now'); if (secs < 3600) return Math.floor(secs / 60) + ' min'; if (secs < 86400) return Math.floor(secs / 3600) + ' h'; return Math.floor(secs / 86400) + ' d'; } function _svcIn(secs) { if (secs == null) return '—'; if (secs <= 0) return t('svc_now'); if (secs < 60) return secs + ' s'; if (secs < 3600) return Math.floor(secs / 60) + ' min'; return Math.floor(secs / 3600) + ' h'; } function _svcRow(label, value) { return `
${esc(label)}${value}
`; } // Render one detail row from its backend descriptor. Each service emits its own // rows in status().detail = [{label_key, value | value_key, fmt?}], so this stays // fully generic — a new service's rows show up with no change here: // value_key → an i18n value (e.g. "Embedded"); fmt 'ago'/'in' → relative time; // otherwise the value is shown verbatim (escaped). function _svcDetailRow(r) { let v; if (r.value_key) v = esc(t(r.value_key)); else if (r.fmt === 'ago') v = _svcAgo(r.value); else if (r.fmt === 'in') v = _svcIn(r.value); else v = esc(r.value == null ? '—' : String(r.value)); return _svcRow(t(r.label_key), v); } // Render one service card. `meta` = {key, icon, title, rows[], canControl, running} function _svcCard(meta) { const cls = _SVC_STATE_CLS[meta.state] || 'secondary'; const canCtl = currentUser.permissions?.includes('services_control'); let actions = ''; if (meta.controllable && canCtl) { actions = meta.running ? `` : ``; } else if (!meta.embedded) { actions = `${esc(t('svc_external_managed'))}`; } return `
${esc(meta.title)}
${esc(_svcStateLabel(meta.state))}
${meta.rows.join('')}
${actions}
`; } async function renderServices() { const c = document.getElementById('services-container'); if (!c) return; const resp = await apiGet('/api/v1/services'); const s = (resp && resp.services) || {}; // Render one card per registered service, in registry order. Identity // (title/icon) + state/controllability come from the backend entry; the // service-specific detail rows from _SVC_DETAIL (or none for an unknown key). const cards = Object.entries(s).map(([key, d]) => _svcCard({ key, icon: d.icon || 'bi-gear', title: d.label_key ? t(d.label_key) : key, state: d.state, running: !!d.running, controllable: !!d.controllable, embedded: !!d.embedded, rows: (d.detail || []).map(_svcDetailRow), })); const refreshBtn = ``; c.innerHTML = `
${esc(t('tab_services'))}
${refreshBtn}
${cards.join('') || `

${esc(t('svc_none'))}

`}
`; // Light auto-refresh while the Services tab stays visible. if (_svcPollTimer) clearTimeout(_svcPollTimer); _svcPollTimer = setTimeout(() => { if (document.getElementById('tab-services')?.classList.contains('active')) renderServices(); }, 5000); } async function servicesControl(name, action) { const r = await apiPost(`/api/v1/services/${name}/${action}`, {}); if (r && (r.status === 200) && r.data?.ok) { showToast(t('svc_' + action + '_ok'), 'success'); } else { const reason = r?.data?.reason; showToast(reason ? (t('svc_err_' + reason) || reason) : t('svc_action_error'), 'warning'); } renderServices(); }
