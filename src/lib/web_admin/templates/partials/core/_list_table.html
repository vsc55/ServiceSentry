{# ================================================================ Generic schema-driven list table. createListTable(spec) wires the shared skeleton every entity table needs — card shell, header (title/accent/chooser/refresh/extra buttons/bulk bar), top+bottom pagination, sortable/reorderable/ resizable/content-fit columns, selection and the actions column — from a per-table SPEC. Each table only declares its columns, cell renderer, actions, header buttons and data source. spec.columns(ctx) : spec.columns); const _defVis = (ctx) => (typeof spec.defaultVisible === 'function' ? spec.defaultVisible(ctx) : spec.defaultVisible); const _readW = () => { try { return JSON.parse(localStorage.getItem(lsW) || '{}'); } catch (_) { return {}; } }; // ── Handlers (published under legacy names) ───────────────────────────── function toggleCol(col) { const vis = st.visible.get(); const adding = !vis.has(col); vis.has(col) ? spec.resetVisible() : _defVis())); st.order.set(_cols().map(c => c.id)); localStorage.setItem(lsCols, JSON.stringify([...st.visible.get()])); localStorage.setItem(lsOrder, JSON.stringify(st.order.get())); localStorage.removeItem(lsW); _persistTableConfig(); render(); } function saveOrder() { localStorage.setItem(lsOrder, JSON.stringify(st.order.get())); _persistTableConfig(); } function goPage(p) { st.page.set(p); render(); } function changePageSize(v) { st.pageSize.set(parseInt(v, 10)); st.page.set(1); render(); } function sortBy(col) { if (st.sort.get() === col) st.sortDir.set(st.sortDir.get() === 'asc' ? 'desc' : 'asc'); else { st.sort.set(col); st.sortDir.set('asc'); } render(); } function selectAll(checked) { const sel = st.selected && st.selected(); if (!sel) return; (st.currentPage.get() || []).forEach(k => { if (spec.selectableRow && !spec.selectableRow(k)) return; // e.g. built-in roles checked ? sel.add(k) : sel.delete(k); }); render(); } function toggleOne(key, checked) { const sel = st.selected && st.selected(); if (!sel) return; checked ? sel.add(key) : sel.delete(key); render(); } function initFeatures() { _attachColFeatures(tableId, () => st.order.get(), o => st.order.set(o), saveOrder, render, (col, w) => { const cw = _readW(); cw[col] = w; localStorage.setItem(lsW, JSON.stringify(cw)); }); } // ── Render ────────────────────────────────────────────────────────────── async function render() { const c = document.getElementById(spec.containerId); if (!c) return; // Full-height layout: the container fills its pane (.ss-vfill), the card // inside it stretches (.ss-vfill) and the table body scrolls (.ss-vscroll) — // the same generic classes every table uses (see web_admin.css). (va - vb) : String(va).localeCompare(String(vb)); return dir === 'asc' ? cmp : -cmp; }); const total = rows.length; const sel = st.selected ? st.selected() : null; // Optional alternate body modes (e.g. sessions card/table toggle). const mode = spec.bodyMode ? spec.bodyMode(ctx) : 'table'; const showChooser = spec.showChooser ? spec.showChooser(mode) : true; // Card header — shared by the empty and populated states. const chooser = showChooser ? _buildColChooser(spec.colChooserCols ? spec.colChooserCols(cols) : cols, st.visible.get(), gp.toggleCol, gp.resetCols, spec.colChooserMinWidth || 160) : ''; const bulkBar = (spec.bulkBar && sel) ? spec.bulkBar(ctx, sel) : ''; const headerLead = spec.headerLead ? (spec.headerLead(ctx, mode) || '') : ''; const refresh = spec.refreshButton ? (spec.refreshButton(ctx) || '') : ''; const headerBtns = spec.headerButtons ? (spec.headerButtons(ctx) || '') : ''; const accentDiv = spec.accent ? `
` : ''; const headerHtml = `
${spec.title(ctx)}
${bulkBar} ${headerLead} ${chooser} ${refresh} ${headerBtns}
`; const cardClass = spec.cardClass || 'ss-card'; if (!total) { // emptyBody → card + header + body; emptyState → bare (no card). // The empty card still fills its pane (.ss-vfill + .ss-vscroll body) so an // empty table stretches exactly like a populated one — no short card. if (spec.emptyBody) c.innerHTML = `
${accentDiv}${headerHtml}
${spec.emptyBody(ctx)}
`; else c.innerHTML = spec.emptyState ? spec.emptyState(ctx) : ''; return; } const ps = st.pageSize.get() || total; const page = Math.max(1, Math.min(st.page.get(), Math.ceil(total / ps))); st.page.set(page); const pageRows = rows.slice((page - 1) * ps, page * ps); st.currentPage.set(pageRows.map(r => spec.rowKey(r))); let h; if (mode === 'cards' && spec.cardsBody) { // Alternate (non-table) body — e.g. the sessions card grid. Selection // checkboxes inside it call the same toggleOne; no columns/resize. h = spec.cardsBody(pageRows, ctx); } else { // Ordered, visible columns. const knownIds = new Set(cols.map(co => co.id)); const order = st.order.get().slice(); cols.forEach(co => { if (!order.includes(co.id)) order.push(co.id); }); st.order.set(order.filter(id => knownIds.has(id))); const orderedCols = st.order.get().map(id => cols.find(co => co.id === id)) .filter(co => co && (co.always || st.visible.get().has(co.id))); const cw = _readW(); const thW = id => cw[id] ? ` style="width:${cw[id]}"` : ''; const thClassMap = spec.thClass || {}; const thRow = orderedCols.map(co => { const inner = co.sortKey ? _thSortInner(co.label(), co.sortKey, st.sort.get(), st.sortDir.get(), gp.sort) : `${esc(co.label())}`; const xc = co.thClass || thClassMap[co.id]; // resizable:false → content-fit header (width:1% + nowrap): not resizable, // sized to its content, still sortable + reorderable. if (co.resizable === false) { const cls = ['ss-th', 'ss-th-fit']; if (xc) cls.push(xc); return `${inner}`; } const cls = ['ss-th', 'ss-th-resizable']; if (xc) cls.push(xc); return `${inner}`; }).join(''); const selectable = (typeof spec.selectable === 'function') ? spec.selectable(ctx) : spec.selectable; const allSel = selectable && sel && pageRows.length > 0 && pageRows.every(r => sel.has(spec.rowKey(r))); const actionsOn = !!spec.actions && (spec.actionsVisible ? spec.actionsVisible(ctx) : true); h = ` ${selectable ? `` : ''} ${thRow} ${actionsOn ? `` : ''} `; for (const row of pageRows) { const k = spec.rowKey(row); const cells = orderedCols.map(co => { // Shared metadata cells (uid/created_at/updated_at/updated_by) render // from the single source; everything else from the table's cell(). const cd = (_META_CELL[co.id] && spec.record) ? ${spec.actionsLabel ? spec.actionsLabel() : t('col_actions')}
${spec.actions(row, ctx)}
`; } const pagination = _renderPagination(total, page, st.pageSize.get(), gp.goPage, gp.changePageSize); c.innerHTML = `
${accentDiv} ${headerHtml}
${pagination}
${h}
${pagination}
`; if (mode !== 'cards') setTimeout(initFeatures, 0); } catch (err) { // Never leave the tab blank on a render/prepare error: log it and show a // minimal empty state (e.g. a table whose emptyBody/prepare threw). console.error(`list-table[${spec.key}] render failed:`, err); if (!c.innerHTML.trim()) c.innerHTML = `
${esc(t('no_items') || 'No items')}
`; } } // ── Publish legacy global names ───────────────────────────────────────── const pub = (name, fn) => { if (name) window[name] = fn; }; pub(gp.render, render); pub(gp.goPage, goPage); pub(gp.changePageSize, changePageSize); pub(gp.sort, sortBy); pub(gp.toggleCol, toggleCol); pub(gp.resetCols, resetCols); pub(gp.saveOrder, saveOrder); pub(gp.selectAll, selectAll); pub(gp.toggleOne, toggleOne); pub(gp.initFeatures, initFeatures); // ── Join the persistence/customisations registry ─────────────────────── if (spec.persist) { _LIST_TABLE_REGISTRY.push({ k: spec.key, def: () => new Set(_defVis()), all: () => _cols(), getVis: () => st.visible.get(), setVis: s => st.visible.set(s), getOrd: () => st.order.get(), setOrd: o => st.order.set(o), ls: [lsCols, lsOrder, lsW], clearLabelKey: spec.clearLabelKey, umCustomised: spec.umCustomised || null, extra: spec.persistExtra || null, // extra table_config fields (e.g. sessions view mode) }); } return { render, goPage, changePageSize, sort: sortBy, toggleCol, resetCols, saveOrder, selectAll, toggleOne, initFeatures }; }
