Defaults to success for backward compatibility. const _lv = { success: 'bi-check-circle', warning: 'bi-exclamation-triangle', danger: 'bi-x-circle' }; const _vc = _lv[variant] ? variant : 'success'; const body = (lead ? `

${esc(lead)}

` : '') + (pairs.length ? `${rows}
` : `

${esc(t('no_items') || 'No data')}

`); document.getElementById('infoModalBody').innerHTML = body; bootstrap.Modal.getOrCreateInstance(document.getElementById('infoModal')).show(); }
