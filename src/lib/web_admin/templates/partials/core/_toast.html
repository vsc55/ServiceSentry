{# ================================================================ Toast notification and generic confirm modal helpers. ================================================================ #} // ──────────────────────────── Toast ─────────────────────────────── // ──────────────────────────── Generic confirm modal ─────────────── let _confirmModalCallback = null; function showConfirmModal(msg, callback) { _confirmModalCallback = callback; document.getElementById('confirmModalMsg').textContent = msg; bootstrap.Modal.getOrCreateInstance(document.getElementById('confirmModal')).show(); } function showToast(msg, type = 'success', delay = 3000) { const t = document.getElementById('toast'); const b = document.getElementById('toast-body'); t.className = 'toast align-items-center border-0 text-bg-' + type; b.textContent = msg; bootstrap.Toast.getOrCreateInstance(t, { delay }).show(); } // ──────────────────────────── Generic info modal ────────────────── // Show a title + an optional lead message + a key/value table. `data` may be a // plain object ({key: value}) or an array of [key, value] pairs (preserves order). function showInfoModal(title, data, lead) { document.getElementById('infoModalTitle').textContent = title || ''; const pairs = Array.isArray(data) ? data : Object.entries(data || {}); const rows = pairs.map(([k, v]) => `${esc(String(k))}` + `${esc(v === null || v === undefined ? '' : String(v))}` ).join(''); const body = (lead ? `

${esc(lead)}

` : '') + (pairs.length ? `${rows}
` : `

${esc(t('no_items') || 'No data')}

`); document.getElementById('infoModalBody').innerHTML = body; bootstrap.Modal.getOrCreateInstance(document.getElementById('infoModal')).show(); }
