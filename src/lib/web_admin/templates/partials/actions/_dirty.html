{# ================================================================ Dirty-state tracking and the unsaved-changes modal. ================================================================ #} // ──────────────────────────── Actions ───────────────────────────── /** Disable all interactive elements in *container* except buttons marked * with data-test-btn (test / action buttons that must remain usable). */ function _applyReadonly(container) { container.querySelectorAll('input, select, textarea').forEach(el => { el.disabled = true; }); container.querySelectorAll('button:not([data-test-btn])').forEach(el => { el.disabled = true; }); } function markDirty(target) { const saved = target === 'modules' ? _savedModulesData : _serverConfigData; const current = target === 'modules' ? modulesData : configData; if (JSON.stringify(current) === JSON.stringify(saved)) { clearDirty(target); return; } const badgeId = target === 'modules' ? 'badgeModulesDirty' : 'badgeConfigDirty'; const btnId = target === 'modules' ? 'btnSaveModules' : 'btnSaveConfig'; document.getElementById(badgeId)?.classList.remove('d-none'); document.getElementById(btnId)?.removeAttribute('disabled'); if (target === 'modules') document.getElementById('btnUndoModules')?.classList.remove('d-none'); } function clearDirty(target) { const badgeId = target === 'modules' ? 'badgeModulesDirty' : 'badgeConfigDirty'; const btnId = target === 'modules' ? 'btnSaveModules' : 'btnSaveConfig'; document.getElementById(badgeId)?.classList.add('d-none'); document.getElementById(btnId)?.setAttribute('disabled', ''); if (target === 'modules') document.getElementById('btnUndoModules')?.classList.add('d-none'); if (target === 'config') { _dirtyFields.clear(); _configConflicts = {}; _serverChangedDirtyPaths.clear(); } } function _isDirty(target) { const id = target === 'modules' ? 'badgeModulesDirty' : 'badgeConfigDirty'; const badge = document.getElementById(id); // No badge at all = that section is not part of this page (standalone pages render a // single pane), so there is nothing unsaved to lose. Note `!el?.x` would yield `true` // here — reading "dirty" — and pop the browser's leave-confirmation on every nav. return !!badge && !badge.classList.contains('d-none'); } /** Show the 3-button unsaved-changes modal (Cancel / Discard / Save). * After Discard or Save, *next* is performed: either a tab button to switch to, or a * callback (used to leave the panel for a standalone section page, where "next" is a * navigation rather than a tab switch). */ function showUnsavedChangesModal(dirtyTarget, next) { document.getElementById('unsavedChangesMsg').textContent = t('unsaved_changes_leave'); const modal = bootstrap.Modal.getOrCreateInstance( document.getElementById('unsavedChangesModal')); function _dismiss() { modal.hide(); } function _go() { if (typeof next === 'function') next(); else bootstrap.Tab.getOrCreateInstance(next).show(); } const btnCancel = document.getElementById('btnUnsavedCancel'); const btnDiscard = document.getElementById('btnUnsavedDiscard'); const btnSave = document.getElementById('btnUnsavedSave'); // Clone nodes to remove any previous listeners const newCancel = btnCancel.cloneNode(true); const newDiscard = btnDiscard.cloneNode(true); const newSave = btnSave.cloneNode(true); btnCancel.replaceWith(newCancel); btnDiscard.replaceWith(newDiscard); btnSave.replaceWith(newSave); newCancel.addEventListener('click', () => _dismiss()); newDiscard.addEventListener('click', () => { if (dirtyTarget === 'modules') { modulesData = deepClone(_savedModulesData); clearDirty('modules'); _renderModulesSurface(); } else { configData = deepClone(_serverConfigData); clearDirty('config'); // also clears _dirtyFields, _configConflicts, _serverChangedDirtyPaths renderConfig(); } _dismiss(); showToast(t('changes_discarded')); _go(); }); newSave.addEventListener('click', async () => { if (dirtyTarget === 'modules') await saveModules(); else await saveConfig(); _dismiss(); if (!_isDirty(dirtyTarget)) _go(); // only navigate if save succeeded }); // Keyboard shortcuts: Escape=Cancel, Enter=Save, Delete=Discard function _onKey(e) { if (e.key === 'Escape') { e.preventDefault(); newCancel.click(); } if (e.key === 'Enter') { e.preventDefault(); newSave.click(); } if (e.key === 'Delete') { e.preventDefault(); newDiscard.click(); } } const modalEl = document.getElementById('unsavedChangesModal'); modalEl.addEventListener('keydown', _onKey); modalEl.addEventListener('hidden.bs.modal', () => { modalEl.removeEventListener('keydown', _onKey); }, { once: true }); modal.show(); }
